# 撤回功能接口测试

## 测试逻辑

1. 用户A发起审批 → 第一个节点"填报人提交"自动完成
2. 进入第二个节点"XX负责人审批"（待审批状态）
3. **在第二个节点审批人处理之前**，用户A可以撤回
4. 撤回后，审批实例状态变为已撤回，Plan可以继续编辑

## 接口测试脚本

### 测试1: 完整撤回流程

```bash
#!/bin/bash

# ==================== 步骤1: 获取Token ====================
echo "=== 步骤1: 登录获取Token ==="
TOKEN=$(curl -s -X POST "http://localhost:8080/api/v1/auth/login" \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"admin123"}' | \
  python3 -c "import sys,json; d=json.load(sys.stdin); print(d['data']['accessToken'])")

echo "Token: ${TOKEN:0:50}..."

# ==================== 步骤2: 创建测试Plan ====================
echo ""
echo "=== 步骤2: 创建测试Plan ==="
PLAN_RESPONSE=$(curl -s -X POST "http://localhost:8080/api/v1/plans" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "planName": "撤回测试Plan",
    "planType": "STRAT_TO_FUNC",
    "targetOrgId": 37,
    "year": 2026,
    "cycleId": 4
  }')

echo "$PLAN_RESPONSE" | python3 -m json.tool

PLAN_ID=$(echo "$PLAN_RESPONSE" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d['data']['id'])" 2>/dev/null)
echo "创建的Plan ID: $PLAN_ID"

# ==================== 步骤3: 启动审批流程 ====================
echo ""
echo "=== 步骤3: 启动审批流程 ==="
WORKFLOW_RESPONSE=$(curl -s -X POST "http://localhost:8080/api/v1/workflows/start" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d "{
    \"workflowCode\": \"PLAN_DISPATCH_STRATEGY\",
    \"businessEntityId\": $PLAN_ID,
    \"businessEntityType\": \"PLAN\"
  }")

echo "$WORKFLOW_RESPONSE" | python3 -m json.tool

INSTANCE_ID=$(echo "$WORKFLOW_RESPONSE" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d['data']['instanceId'])" 2>/dev/null)
echo "创建的审批实例ID: $INSTANCE_ID"

# ==================== 步骤4: 检查实例状态（确认可以撤回） ====================
echo ""
echo "=== 步骤4: 检查实例状态 ==="
INSTANCE_DETAIL=$(curl -s -X GET "http://localhost:8080/api/v1/workflows/instances/$INSTANCE_ID" \
  -H "Authorization: Bearer $TOKEN")

echo "$INSTANCE_DETAIL" | python3 -c "
import sys, json
d = json.load(sys.stdin)
data = d.get('data', {})
print(f\"实例状态: {data.get('status')}\")
print(f\"发起人ID: {data.get('starterId')}\")
print(f\"当前节点: {data.get('currentStepName')}\")
print(f\"当前审批人: {data.get('currentApproverName')}\")
print(f\"是否可撤回: {data.get('canWithdraw')}\")
print(\"\n任务列表:\")
for task in data.get('tasks', []):
    print(f\"  - {task.get('taskName')}: {task.get('status')} (操作人: {task.get('assigneeName')})\")
"

CAN_WITHDRAW=$(echo "$INSTANCE_DETAIL" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('data',{}).get('canWithdraw'))" 2>/dev/null)

if [ "$CAN_WITHDRAW" != "true" ]; then
    echo ""
    echo "❌ 测试失败: 实例状态显示不可撤回"
    exit 1
fi

# ==================== 步骤5: 执行撤回操作 ====================
echo ""
echo "=== 步骤5: 执行撤回操作 ==="
CANCEL_RESPONSE=$(curl -s -X POST "http://localhost:8080/api/v1/workflows/$INSTANCE_ID/cancel" \
  -H "Authorization: Bearer $TOKEN")

echo "$CANCEL_RESPONSE" | python3 -m json.tool

CANCEL_SUCCESS=$(echo "$CANCEL_RESPONSE" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('success'))" 2>/dev/null)

if [ "$CANCEL_SUCCESS" != "True" ]; then
    echo ""
    echo "❌ 测试失败: 撤回操作失败"
    exit 1
fi

# ==================== 步骤6: 验证撤回后的状态 ====================
echo ""
echo "=== 步骤6: 验证撤回后的状态 ==="
INSTANCE_AFTER=$(curl -s -X GET "http://localhost:8080/api/v1/workflows/instances/$INSTANCE_ID" \
  -H "Authorization: Bearer $TOKEN")

echo "$INSTANCE_AFTER" | python3 -c "
import sys, json
d = json.load(sys.stdin)
data = d.get('data', {})
print(f\"实例状态: {data.get('status')}\")
print(f\"是否可撤回: {data.get('canWithdraw')}\")
"

INSTANCE_STATUS=$(echo "$INSTANCE_AFTER" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('data',{}).get('status'))" 2>/dev/null)

if [ "$INSTANCE_STATUS" == "REJECTED" ] || [ "$INSTANCE_STATUS" == "WITHDRAWN" ] || [ "$INSTANCE_STATUS" == "CANCELLED" ]; then
    echo ""
    echo "✅ 测试通过: 撤回成功，实例状态已变更为 $INSTANCE_STATUS"
else
    echo ""
    echo "⚠️ 测试警告: 撤回后实例状态为 $INSTANCE_STATUS"
fi

# ==================== 步骤7: 验证Plan可以继续编辑 ====================
echo ""
echo "=== 步骤7: 验证Plan可以继续编辑 ==="
PLAN_AFTER=$(curl -s -X GET "http://localhost:8080/api/v1/plans/$PLAN_ID" \
  -H "Authorization: Bearer $TOKEN")

echo "$PLAN_AFTER" | python3 -c "
import sys, json
d = json.load(sys.stdin)
data = d.get('data', {})
print(f\"Plan状态: {data.get('status')}\")
print(f\"工作流实例ID: {data.get('workflowInstanceId')}\")
print(f\"可编辑: {data.get('canEdit')}\")
"

echo ""
echo "=== 测试完成 ==="
```

### 测试2: 第二个节点审批后不能撤回

```bash
#!/bin/bash

TOKEN=$(curl -s -X POST "http://localhost:8080/api/v1/auth/login" \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"admin123"}' | \
  python3 -c "import sys,json; d=json.load(sys.stdin); print(d['data']['accessToken'])")

  -H "Authorization: Bearer $TOKEN" | python3 -c "

data = d.get('data', {})

print(f\"当前状态: {data.get('status')}\")

print(f\"是否可撤回: {data.get('canWithdraw')}\")
print(\"\n任务列表:\")
for task in tasks:
    print(f\"  {task.get('taskKey')}: {task.get('status')} - {task.get('taskName')}\")
"

# 获取当前任务ID
CURRENT_TASK_ID=$(curl -s -X GET "http://localhost:8080/api/v1/workflows/instances/$INSTANCE_ID" \
  -H "Authorization: Bearer $TOKEN" | \
  python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('data',{}).get('currentTaskId'))" 2>/dev/null)

  -d '{"comment":"同意"}')

  -H "Authorization: Bearer $TOKEN")

if d.get('success') == False:
    print(f\"✅ 测试通过: 撤回被拒绝，原因: {d.get('message')}\")
else:
    print(f\"❌ 测试失败: 审批后仍然可以撤回\")
```

## 预期结果

### 场景1: 正常撤回
- 发起人提交后 → 实例状态 `IN_REVIEW`
- 第二个节点待审批 → `canWithdraw: true`
- 执行撤回 → 返回 `success: true`
- 撤回后 → 实例状态 `REJECTED`/`WITHDRAWN`/`CANCELLED`，`canWithdraw: false`

### 场景2: 审批后不能撤回
- 第二个节点审批通过 → 进入第三个节点
- 尝试撤回 → 返回 `success: false`，错误信息提示不允许撤回

### 场景3: 非发起人不能撤回
- 非发起人尝试撤回 → 返回 `success: false`，错误信息提示无权限
